<script>
// Typing message
let text = "I know you're angry with me... and I truly deserve it. But I never wanted to hurt you. You mean so much to me... more than I can explain. I'm really sorry. If you can, please give me one chance to fix everything.";

let i = 0;
function typing() {
    if (i < text.length) {
        document.getElementById("message").innerHTML += text.charAt(i);
        i++;
        setTimeout(typing, 35);
    }
}
typing();

// Play music
function playMusic() {
    document.getElementById("song").play();
}

// Forgive popup
function forgive() {
    alert("Thank you... this means everything to me ❤️");
}

// Floating hearts
function createHeart() {
    let heart = document.createElement("div");
    heart.className = "heart";
    heart.innerHTML = "❤️";
    heart.style.left = Math.random() * 100 + "vw";
    heart.style.fontSize = Math.random() * 20 + 15 + "px";
    document.body.appendChild(heart);

    setTimeout(() => {
        heart.remove();
    }, 6000);
}

setInterval(createHeart, 300);
</script>
